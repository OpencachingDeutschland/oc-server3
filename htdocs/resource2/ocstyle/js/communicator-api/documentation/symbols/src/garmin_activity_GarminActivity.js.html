 21 /**A data structure for storing data commonly found in various
 22  * formats supported by various gps devices.  Some examples are
 23  * gpx track, gpx route, gpx wayoint, and tcx activity.
 24  * @class Garmin.Activity
 25  * @constructor 
 26  */
 27 Garmin.Activity = function(){};
 28 Garmin.Activity = Class.create();
 29 Garmin.Activity.prototype = {
 30 	
 31 	initialize: function() {
 32 		this.attributes = new Hash();
 33 		this.summary = new Garmin.Sample();
 34 		this.series = new Array();
 35 	},
 36 	
 37 	getAttributes: function() {
 38 		return this.attributes;
 39 	},
 40 	
 41 	getAttribute: function(aKey) {
 42 		return this.attributes[aKey];
 43 	},
 44 	
 45 	setAttribute: function(aKey, aValue) {
 46 		this.attributes[aKey] = aValue;
 47 	},
 48 	
 49 	getSeries: function() {
 50 		return this.series;
 51 	},
 52 	
 53 	getHistorySeries: function() {
 54 		for (var i = 0; i < this.series.length; i++) {
 55 			if (this.series[i].getSeriesType() == Garmin.Series.TYPES.history) {
 56 				return this.series[i];
 57 			}
 58 		}
 59 		return null;
 60 	},
 61 	
 62 	addSeries: function(series) {
 63 		this.series.push(series);
 64 	},
 65 	
 66 	getSingleSeries: function(index) {
 67 		var targetSeries = null;
 68 		if (index >= 0 && index < this.series.length) {
 69 			targetSeries = this.series[index];
 70 		}
 71 		return targetSeries;
 72 	},
 73 	
 74 	getSummary: function() {
 75 		return this.summary;
 76 	},
 77 	
 78 	getSummaryValue: function(sKey) {
 79 		return this.summary.getMeasurement(sKey);
 80 	},
 81 	
 82 	setSummaryValue: function(sKey, sValue, sContext) {
 83 		this.summary.setMeasurement(sKey, sValue, sContext);
 84 	},
 85 	
 86 	getEndTime: function() {
 87 		return this.getSummaryValue(Garmin.Activity.SUMMARY_KEYS.endTime).getValue();
 88 	},	
 89 	
 90 	getStartTime: function() {
 91 		return this.getSummaryValue(Garmin.Activity.SUMMARY_KEYS.startTime).getValue();
 92 	},
 93 	
 94 	printMe: function(tabs) {
 95 		var output = "";
 96 		output += tabs + "\n\n[Activity]\n";
 97 		
 98 		output += tabs + "  attributes:\n";
 99 		var attKeys = this.attributes.keys();
100 		for (var i = 0; i < attKeys.length; i++) {
101 			output += tabs + "    " + attKeys[i] + ": " + this.attributes[attKeys[i]] + "\n"; 
102 		}
103 		
104 		output += tabs + "  summary:\n";
105 		output += this.summary.printMe(tabs + "  ");
106 
107 		output += tabs + "  series:\n";		
108 		for (var i = 0; i < this.series.length; i++) {
109 			output += this.series[i].printMe(tabs + "  ");
110 		}
111 		
112 		return output;
113 	},
114 	
115 	toString: function() {
116 		return "[Garmin.Activity]"
117 	}
118 };
