 21 /** The newer activity filter for synchronizing lists.  It makes an AJAX request
 22  * to a remote service in order to detect matches on the server.  It is dependent
 23  * on Garmin.Axm.ActivityMatch, which wraps around the json response making it 
 24  * easy to retrieve the JSON values via javascript.   
 25  *  
 26  * @class Garmin.ActivityMatcher
 27  */ 
 28 Garmin.ActivityMatcher = function(deviceXml, allActivityIds, ajaxUrl, ajaxOptions, callback){}; //just here for jsdoc
 29 Garmin.ActivityMatcher = Class.create();
 30 Garmin.ActivityMatcher.prototype = {
 31     initialize: function(deviceXml, allActivityIds, ajaxUrl, ajaxOptions, callback) {
 32         this.allActivityIds = allActivityIds;
 33         this.ajaxUrl = ajaxUrl;
 34         this.ajaxOptions = ajaxOptions;
 35         this.ajaxOptions.parameters = {deviceXml: deviceXml, externalIds: allActivityIds};
 36         this.callback = callback;
 37         
 38         this.apiRequest = new Garmin.RemoteTransfer();
 39         this.apiResponse = null;
 40         
 41         this.activityMatches = null;
 42     },
 43     
 44     /** Run the filter.  Makes the AJAX request to find matching IDs on the
 45      * server and populates the filtered list. 
 46      */
 47     run: function() {
 48         
 49         this.ajaxOptions.onSuccess = function(xhr) {
 50             this.activityMatches = new Garmin.Axm.ActivityMatch(xhr.responseJSON);
 51         	this.callback();
 52     	}.bind(this);
 53         
 54         this.apiResponse = this.apiRequest.openRequest(this.ajaxUrl, this.ajaxOptions);
 55     },
 56     
 57     /** Get the matching JSON object for given activity ID from the original response object.
 58      * Null if not found or if activity matcher service is unavailable.
 59      */
 60     get: function(activityId) {
 61         if( this.activityMatches != null) {
 62             return this.activityMatches.getMatch(activityId);
 63         }
 64     }
 65 }
 66 
 67 /**
 68  * Wrapper for the Axm activityMatch JSON return object.  This is a lightweight wrapper.  All
 69  * it does is find the given activity in the original response object and return it.  You should access
 70  * the properties of that object directly.
 71  */
 72 if (Garmin.Axm == undefined) Garmin.Axm = {};
 73 Garmin.Axm.ActivityMatch = function(){}; // for jsdoc
 74 Garmin.Axm.ActivityMatch = Class.create();
 75 Garmin.Axm.ActivityMatch.prototype = {
 76     
 77     initialize: function(json) {
 78         if( json != null && json['matches'] != null) {
 79             this.activityMatchesJson = json['matches'];
 80         }
 81     },
 82     
 83     /** Looks up the match for the given external ID.  This returns the JSON object for 
 84      * direct access.  isMatch and isDeleted can be accessed directly from the return object:
 85      * 
 86      * returned.isMatch
 87      * returned.isDeleted
 88      * 
 89      * @return {JSON} the match result, or null if the ID is not found in the original response
 90      */
 91     getMatch: function(activityId) {
 92         return this.activityMatchesJson[activityId];
 93     }
 94 }
