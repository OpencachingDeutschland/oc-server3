 20 /**Contains a series of samples.  Could represent tracks, routes, waypoints, and
 21  * many other types of data.
 22  * @class Garmin.Series
 23  * @constructor 
 24  * @param (String) type - the type of data this series contain.  Should be determined by what
 25  * 							information is recorded by the samples this series contain.
 26  */
 27 Garmin.Series = function(type){};
 28 Garmin.Series = Class.create();
 29 Garmin.Series.prototype = {
 30 
 31 	initialize: function(seriesType) {
 32 		this.seriesType = seriesType;
 33 		this.samples = new Array();
 34 	},
 35 
 36 	getSeriesType: function() {
 37 		return this.seriesType;
 38 	},
 39 
 40 	setSeriesType: function(seriesType) {
 41 		this.seriesType = seriesType;
 42 	},
 43 
 44 	getSamples: function() {
 45 		return this.samples;		
 46 	},
 47 	
 48 	getSample: function(index) {
 49 		var targetSample = null;
 50 		if (index >= 0 && index < this.samples.length) {
 51 			targetSample = this.samples[index];
 52 		}
 53 		return targetSample;
 54 	},
 55 	
 56 	addSample: function(sample) {
 57 		if (sample != null) {
 58 			this.samples.push(sample);			
 59 		}
 60 	},
 61 	
 62 	getSamplesLength: function() {
 63 		return this.samples.length;
 64 	},
 65 
 66 	getFirstValidLocationSample: function() {
 67 		return this.findNearestValidLocationSample(0, 1);
 68 	},
 69 	
 70 	getLastValidLocationSample: function() {
 71 		return this.findNearestValidLocationSample(this.getSamplesLength()-1, -1);
 72 	},
 73     
 74     /** Find the nearest valid location point to the index given
 75      * 
 76      * @param index is the index
 77      * @param incDirection is an int in the direction we'd like to look positive 
 78      * 	nums are forward, negative nums are backwards
 79      * 
 80      * @type Garmin.Sample 
 81      * @return The nearest point (possibly the index) that has a valid latitude and longitude
 82      */ 
 83     findNearestValidLocationSample: function(index, incDirection) {
 84 		return this._findNearestValidLocationSampleInternal(index, incDirection, 0);
 85     },
 86 	
 87 	_findNearestValidLocationSampleInternal: function(index, incDirection, count) {
 88 		// make sure we haven't looped through every element already
 89 		if (this.getSamplesLength() > 0 && count < this.getSamplesLength()) {
 90 			// make sure index requested is within bounds
 91 			if (index >= 0 && index < this.getSamplesLength()) {
 92 				var sample = this.getSample(index);
 93 				if (sample.isValidLocation()) {
 94 					return sample;	
 95 				} else {
 96 					return this._findNearestValidLocationSampleInternal(index + incDirection, incDirection, ++count);
 97 				}
 98 			} else if (index > this.getSamplesLength()) {
 99 				return this._findNearestValidLocationSampleInternal(this.getSamplesLength()-1, -1, count);
100 			} else {
101 				return this._findNearestValidLocationSampleInternal(0, 1, count);
102 			}	
103 		} else {
104 			return null;
105 		}		
106 	},
107 	
108 	printMe: function(tabs) {
109 		var output = tabs + "  [Series]\n";
110 		output += tabs + "    seriesType: " + this.seriesType + "\n";	
111 		output += tabs + "    samples:\n";
112 		for (var i = 0; i < this.samples.length; i++) {
113 			output += this.samples[i].printMe(tabs + "    ");
114 		}
115 		return output;
116 	},
117 	
118 	toString: function() {
119 		return "[Series]";
120 	}
121 };
