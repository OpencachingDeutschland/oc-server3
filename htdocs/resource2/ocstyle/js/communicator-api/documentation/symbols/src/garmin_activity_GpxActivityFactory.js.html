 26 Garmin.GpxActivityFactory = function(){};
 27 Garmin.GpxActivityFactory = {
 28 	
 29 	parseString: function(gpxString) {
 30 		var gpxDocument = Garmin.XmlConverter.toDocument(gpxString);		
 31 		return Garmin.GpxActivityFactory.parseDocument(gpxDocument);	
 32 	},
 33 	
 34 	parseDocument: function(gpxDocument) {
 35         return this.parseDocumentByType(gpxDocument, Garmin.GpxActivityFactory.GPX_TYPE.all);
 36 	},
 37 	
 38 	parseDocumentByType: function(gpxDocument, type) {
 39 		var activities = new Array();
 40         var routes = new Array();
 41         var tracks = new Array();
 42         var waypoints = new Array();        
 43         
 44         switch(type) {
 45           case Garmin.GpxActivityFactory.GPX_TYPE.routes:
 46               activities = Garmin.GpxActivityFactory._parseGpxRoutes(gpxDocument);
 47               break;
 48           case Garmin.GpxActivityFactory.GPX_TYPE.waypoints:
 49               activities = Garmin.GpxActivityFactory._parseGpxWaypoints(gpxDocument);
 50               break;
 51           case Garmin.GpxActivityFactory.GPX_TYPE.tracks:
 52               activities = Garmin.GpxActivityFactory._parseGpxTracks(gpxDocument);
 53               break;
 54           case Garmin.GpxActivityFactory.GPX_TYPE.all:
 55               routes = Garmin.GpxActivityFactory._parseGpxRoutes(gpxDocument);
 56               tracks = Garmin.GpxActivityFactory._parseGpxTracks(gpxDocument);
 57               waypoints = Garmin.GpxActivityFactory._parseGpxWaypoints(gpxDocument);
 58               activities = waypoints.concat(routes).concat(tracks);
 59               break;    
 60         }
 61          
 62         return activities;
 63 	},

 65 	produceString: function(activities) {
 66 		var gpxString = "";
 67 		
 68 		// default creator information incase we can't find the creator info in the dom
 69 		var creator = Garmin.GpxActivityFactory.DETAIL.creator;
 70 		
 71 		// default metadata information incase we can't find the metadata node in the dom
 72 		var metadata = "\n  <metadata>";
 73 		metadata += "\n    <link href=\"http://www.garmin.com\">";
 74 		metadata += "\n      <text>Garmin International</text>";
 75 		metadata += "\n    </link>";						
 76 		metadata += "\n  </metadata>";
 77 		
 78 		// try to find creator and metadata info in the dom
 79 		if (activities != null && activities.length > 0) {
 80 			var activityDom = activities[0].getAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom);
 81 			var gpxNode = activityDom.ownerDocument.getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.gpx);
 82 			if (gpxNode.length > 0) {
 83 				// grab creator information from the dom if possible
 84 				var creatorStr = gpxNode[0].getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.creator);
 85 				if (creatorStr != null && creatorStr != "") {
 86 					creator = creatorStr;
 87 				}
 88 				// grab metadata info
 89 				var metadataNode = gpxNode[0].getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.metadata);
 90 				if (metadataNode.length > 0) {
 91 					metadata = Garmin.XmlConverter.toString(metadataNode[0]);
 92 				}
 93 			}
 94 		}
 95 
 96 		// header tags
 97 		gpxString += "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"no\" ?>";	
 98 		gpxString += "\n<gpx xmlns=\"http://www.topografix.com/GPX/1/1\" creator=\"" + creator + "\" version=\"1.1\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://www.topografix.com/GPX/1/1 http://www.topografix.com/GPX/1/1/gpx.xsd http://www.garmin.com/xmlschemas/GpxExtensions/v3 http://www.garmin.com/xmlschemas/GpxExtensions/v3/GpxExtensionsv3.xsd\">";
 99 	
100 		//metadata tag
101 		gpxString += "\n  " + metadata;
102 		
103 		if (activities != null) {
104 			// waypoint and track tags
105 			for(var i = 0; i < activities.length; i++) {		
106 				gpxString += "\n  " + Garmin.GpxActivityFactory._produceActivityString(activities[i]);
107 			}
108 		}
109 		
110 		// footer tags
111 		gpxString += "\n</gpx>";
112 		
113 		return gpxString;
114 	},

116 	/** Fully load the sample, assume sample was previously lazy-loaded
117 	 */	
118 	finishLoadingSample: function(domNode, sample) {
119 		if (domNode.nodeName == Garmin.GpxActivityFactory.SCHEMA_TAGS.routePoint) {
120 			Garmin.GpxActivityFactory._parseGpxRoutePoint(domNode, sample);
121 			sample.isLazyLoaded = false;
122 		} else if (domNode.nodeName == Garmin.GpxActivityFactory.SCHEMA_TAGS.trackPoint) {
123 			Garmin.GpxActivityFactory._parseGpxTrackPoint(domNode, sample);
124 			sample.isLazyLoaded = false;
125 		}
126 	},

128 	_produceActivityString: function(activity) {
129 		var activityString = "";
130 		if (activity != null) {
131 			var series = activity.getSeries();
132 			for (var i = 0; i < series.length; i++) {
133 				var currentSeries = series[i];
134 				if (currentSeries.getSeriesType() == Garmin.Series.TYPES.history) {
135 					// converting the dom back into string
136 					// this is the lazy way, this will not work if 
137 					// converting between file types or activity data
138 					// has been modified.
139 					var activityDom = activity.getAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom);			
140 					if (activityDom != null) {
141 						activityString = Garmin.XmlConverter.toString(activityDom);
142 					}
143 				} else if (currentSeries.getSeriesType() == Garmin.Series.TYPES.waypoint) {
144 					// converting the dom back into string
145 					// this is the lazy way, this will not work if 
146 					// converting between file types or activity data
147 					// has been modified.
148 					var activityDom = activity.getAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom);			
149 					if (activityDom != null) {
150 						activityString = Garmin.XmlConverter.toString(activityDom);
151 					}						
152 				}
153 			}
154 		}		
155 		return activityString;
156 	},

158 	_parseGpxRoutes: function(gpxDocument) {
159 		var routes = new Array();
160     	var routeNodes = gpxDocument.getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.route);
161 
162 		for( var i=0; i < routeNodes.length; i++ ) {
163 			var route = new Garmin.Activity();
164 			
165 			var routeName = Garmin.GpxActivityFactory._tagValue(routeNodes[i], Garmin.GpxActivityFactory.SCHEMA_TAGS.routeName);
166 			if (routeName == null) {
167 				routeName = "";
168 			}
169 			
170 			route.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom, routeNodes[i]);
171 			route.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.activityName, routeName);
172 
173 			var series = new Garmin.Series(Garmin.Series.TYPES.route);
174 			route.addSeries(series);
175 
176 			var routePoints = routeNodes[i].getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.routePoint);					
177 			if (routePoints.length > 0) {					
178 				for( var j=0; j < routePoints.length; j++ ) {
179 					var routePoint = new Garmin.Sample();
180 					routePoint.setLazyLoading(true, Garmin.GpxActivityFactory, routePoints[j]);
181 					series.addSample(routePoint);
182 				}
183 			}
184 			
185 			if (series.getSamplesLength() > 0) {
186 				routes.push(route);
187 			}
188 		}
189 		
190     	return routes;			
191 	},
192 	
193 	_parseGpxRoutePoint: function(routePointNode, routePointSample) {
194 		if (routePointSample == null) {
195 			routePointSample = new Garmin.Sample();
196 		}
197 
198 		routePointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.latitude, routePointNode.getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointLatitude));
199 		routePointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.longitude, routePointNode.getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointLongitude));
200 		
201 		var elevation =  Garmin.GpxActivityFactory._tagValue(routePointNode,Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointElevation);
202 		if (elevation != null) {
203 			routePointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.elevation, elevation);
204 		}		
205 		
206 		return routePointSample;		
207 	},
208 	
209 	_parseGpxTracks: function(gpxDocument) {
210 		var tracks = new Array();
211 		
212     	var trackNodes = gpxDocument.getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.track);
213 		for( var i=0; i < trackNodes.length; i++ ) {
214 			var track = new Garmin.Activity();
215 			
216 			var trackName = Garmin.GpxActivityFactory._tagValue(trackNodes[i], Garmin.GpxActivityFactory.SCHEMA_TAGS.trackName);
217 			if (trackName == null) {
218 				trackName = "";
219 			}
220 			
221 			track.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom, trackNodes[i]);
222 			track.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.activityName, trackName);
223 
224 			var series = new Garmin.Series(Garmin.Series.TYPES.history);
225 			track.addSeries(series);
226 
227 			var trackSegments = trackNodes[i].getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.trackSegment);	
228 			for( var j=0; j < trackSegments.length; j++ ) {
229 				
230 				// grab all the trackpoints
231 				var trackPoints = trackSegments[j].getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.trackPoint);											
232 				if (trackPoints.length > 0) {
233 					
234 					// set the start and end time summary values		
235 					var startTime = Garmin.GpxActivityFactory._tagValue(trackPoints[0], Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointTime);
236 					var endTime = Garmin.GpxActivityFactory._tagValue(trackPoints[trackPoints.length - 1], Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointTime);					
237 					if (startTime != null && endTime != null) {
238 						track.setSummaryValue(Garmin.Activity.SUMMARY_KEYS.startTime, (new Garmin.DateTimeFormat()).parseXsdDateTime(startTime));
239 						track.setSummaryValue(Garmin.Activity.SUMMARY_KEYS.endTime, (new Garmin.DateTimeFormat()).parseXsdDateTime(endTime));
240 					} else {
241 						// can't find timestamps, must be a route reported as a track (GPSMap does this)
242 						series.setSeriesType(Garmin.Series.TYPES.route);
243 					}
244 				
245 					// loop through all the trackpoints in this segment				
246 					for( var k=0; k < trackPoints.length; k++ ) {
247 						var trackPoint = new Garmin.Sample();
248 						trackPoint.setLazyLoading(true, Garmin.GpxActivityFactory, trackPoints[k]);
249 						series.addSample(trackPoint);						
250 					}
251 					
252 					// add the track to the list of tracks
253 					tracks.push(track);
254 				}
255 			}
256 		}
257 
258     	return tracks;	
259 	},
260 	
261 	_parseGpxTrackPoint: function(trackPointNode, trackPointSample) {
262 		if (trackPointSample == null) {
263 			trackPointSample = new Garmin.Sample();	
264 		}
265 		
266 		trackPointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.latitude, trackPointNode.getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointLatitude));
267 		trackPointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.longitude, trackPointNode.getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointLongitude));
268 		
269 		var elevation =  Garmin.GpxActivityFactory._tagValue(trackPointNode,Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointElevation);
270 		if (elevation != null) {
271 			trackPointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.elevation, elevation);
272 		}
273 
274 		var time = Garmin.GpxActivityFactory._tagValue(trackPointNode, Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointTime);
275 		if (time != null) {
276 			trackPointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.time, (new Garmin.DateTimeFormat()).parseXsdDateTime(time));
277 		}			
278 		
279 		return trackPointSample;
280 	},
281 	
282 	_parseGpxWaypoints: function(gpxDocument) {
283 		var waypoints = new Array();
284     	var waypointNodes = gpxDocument.getElementsByTagName(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypoint);
285     	
286 		for( var i=0; i < waypointNodes.length; i++ ) {
287 			waypoints.push(Garmin.GpxActivityFactory._parseGpxWaypoint(waypointNodes[i]));
288 		}
289     	
290     	return waypoints;
291 	},
292 	
293 	_parseGpxWaypoint: function(waypointNode) {
294 		var waypoint = new Garmin.Activity();
295 		var waypointSeries = new Garmin.Series(Garmin.Series.TYPES.waypoint);
296 		var waypointSample = new Garmin.Sample();
297 		
298 		waypoint.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom, waypointNode);			
299 
300 		waypointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.latitude, waypointNode.getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointLatitude)); 
301 		waypointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.longitude, waypointNode.getAttribute(Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointLongitude));
302 				
303 		var elevation =  Garmin.GpxActivityFactory._tagValue(waypointNode,Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointElevation);
304 		if (elevation != null) {
305 			waypointSample.setMeasurement(Garmin.Sample.MEASUREMENT_KEYS.elevation, elevation);
306 		}
307 		
308 		var wptName =  Garmin.GpxActivityFactory._tagValue(waypointNode,Garmin.GpxActivityFactory.SCHEMA_TAGS.waypointName);
309 		if (wptName != null) {
310 			waypoint.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.activityName, wptName);
311 		}
312 		
313 		waypointSeries.addSample(waypointSample);
314 		waypoint.addSeries(waypointSeries);		
315    		return waypoint;
316 	},
317 	
318 	_tagValue: function(parentNode, tagName) {
319 		var subNode = parentNode.getElementsByTagName(tagName);
320 		return subNode.length > 0 ? subNode[0].childNodes[0].nodeValue : null;
321 	},
