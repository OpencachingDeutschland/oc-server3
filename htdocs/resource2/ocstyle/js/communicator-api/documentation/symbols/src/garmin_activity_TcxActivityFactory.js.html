 34 	/* Creates and returns an array of activities from the document. */
 35 	parseDocument: function(tcxDocument) {
 36 		
 37 		// Not TCX parseable
 38 		if( tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.activities).length == 0
 39 			&& tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.courses).length == 0) {
 40 			throw new Error("ERROR: Unable to parse TCX document.");
 41 		}
 42 		
 43 		var parsedDocument;
 44 		
 45 		// Activities		
 46 		if( tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.activity).length >= 0) {
 47 			
 48 			if( tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.track).length >= 0) { 
 49 				// Complete activity
 50 				parsedDocument = Garmin.TcxActivityFactory._parseTcxActivities(tcxDocument);
 51 			}
 52 			else {
 53 				// Directory listing
 54 				parsedDocument = Garmin.TcxActivityFactory._parseTcxHistoryDirectory(tcxDocument);
 55 			}
 56 		} 
 57 		// Courses
 58 		else if(tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.course).length >= 0) {
 59 		
 60 			if( tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.lap).length >= 0) {
 61 				// Complete course
 62 				parsedDocument = Garmin.TcxActivityFactory._parseTcxCourses(tcxDocument);
 63 			}
 64 			else {
 65 				// Directory listing
 66 				parsedDocument = Garmin.TcxActivityFactory._parseTcxCourseDirectory(tcxDocument);
 67 			}
 68 		}
 69 		
 70 		return parsedDocument;
 71 	},

 73 	produceString: function(activities) {
 74 		var tcxString = "";
 75 		
 76 		// header tags
 77 		tcxString += '<?xml version="1.0" encoding="UTF-8" standalone="no" ?>';
 78 		tcxString += '\n<TrainingCenterDatabase xmlns="http://www.garmin.com/xmlschemas/TrainingCenterDatabase/v2" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://www.garmin.com/xmlschemas/TrainingCenterDatabase/v2 http://www.garmin.com/xmlschemas/TrainingCenterDatabasev2.xsd http://www.garmin.com/xmlschemas/FatCalories/v1 http://www.garmin.com/xmlschemas/fatcalorieextensionv1.xsd">';
 79 		tcxString += '\n  <Activities>';		
 80 		
 81 		if (activities != null && activities.length > 0) {			
 82 			// activity tags
 83 			for (var i = 0; i < activities.length; i++) {
 84 				tcxString += "\n    " + Garmin.TcxActivityFactory._produceActivityString(activities[i]);
 85 			}
 86 			tcxString += '\n  </Activities>';
 87 			
 88 			// author tag
 89 			var activityDom = activities[0].getAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom);
 90 			if (activityDom != null) {
 91 				var authorDom = activityDom.ownerDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.author);
 92 				if (authorDom.length > 0) {
 93 					tcxString += "\n  " + Garmin.XmlConverter.toString(authorDom[0]);
 94 				}
 95 			}
 96 		}
 97 
 98 		// footer tags
 99 		tcxString += '\n</TrainingCenterDatabase>';
100 				
101 		return tcxString;
102 	},

111 	_produceActivityString: function(activity) {
112 		var activityString = "";
113 		
114 		if (activity != null) {
115 			// converting the dom back into string
116 			// this is the lazy way, this will not work if 
117 			// converting between file types or activity data
118 			// has been modified.
119 			var activityDom = activity.getAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom);			
120 			if (activityDom != null) {
121 				activityString = Garmin.XmlConverter.toString(activityDom);
122 			}
123 		}
124 		
125 		return activityString;

128 	_parseTcxHistoryDirectory: function(tcxDocument) {
129 		var activities = new Array();
130 		var activityNodes;
131 
132 		// Grab the activity/course nodes, depending on document		
133 		activityNodes = tcxDocument.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.activity);
134 		
135 		// loop through all activities in the document
136 		for (var i = 0; i < activityNodes.length; i++) {
137 			
138 			if( activityNodes[i].parentNode.tagName != Garmin.TcxActivityFactory.SCHEMA_TAGS.nextSport ){
139 				// create new activity object
140 				var activity = Garmin.TcxActivityFactory._parseTcxActivity(activityNodes[i], Garmin.TcxActivityFactory.SCHEMA_TAGS.activity);
141 				
142 				// grab all the lap nodes in the dom			
143 				var lapNodes = activityNodes[i].getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.lap);
144 				
145 				// grab start time from the first lap and set duration to 0
146 				if (lapNodes.length > 0) {
147 					var activityStartTimeMS = lapNodes[0].getAttribute(Garmin.TcxActivityFactory.SCHEMA_TAGS.lapStartTime);
148 					var activityDurationMS = 0;	// in ms
149 				}			
150 				
151 				// loop through all laps in this activity
152 				for (var j = 0; j < lapNodes.length; j++) {
153 					
154 					// update the duration of this activity
155 					var lapTotalTime = Garmin.TcxActivityFactory._tagValue(lapNodes[j], Garmin.TcxActivityFactory.SCHEMA_TAGS.lapTotalTime);
156 					activityDurationMS += parseFloat(lapTotalTime + "e+3");
157 				}
158 				
159 				if ( lapNodes.length > 0) {
160 					// set the start and end time summary data for the activity if possible
161 					activityStartTimeObj = (new Garmin.DateTimeFormat()).parseXsdDateTime(activityStartTimeMS);
162 					activityEndTimeObj	=  new Garmin.DateTimeFormat();
163 					// NOTE: switch to using setDate() once it is implemented in Garmin.DateTimeFormat
164 					activityEndTimeObj.date = new Date(activityStartTimeObj.getDate().getTime() + activityDurationMS);
165 					activity.setSummaryValue(Garmin.Activity.SUMMARY_KEYS.startTime, activityStartTimeObj);
166 					activity.setSummaryValue(Garmin.Activity.SUMMARY_KEYS.endTime, activityEndTimeObj);
167 				}
168 				
169 				// Add the populated activity to the list of activities.  This activity may not have laps (if it's a directory listing entry).
170 				activities.push(activity);
171 			}
172 		}
173 		
174 		return activities;
175 	},

215 			var historySeries = new Garmin.Series(Garmin.Series.TYPES.history);
216 			
217 			// grab all the lap nodes in the dom			
218 			var lapNodes = activityNodes[i].getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.lap);
219 			
220 			// grab start time from the first lap and set duration to 0
221 			if (lapNodes.length > 0) {
222 				var activityStartTimeMS = lapNodes[0].getAttribute(Garmin.TcxActivityFactory.SCHEMA_TAGS.lapStartTime);
223 				var activityDurationMS = 0;	// in ms
224 			}			
225 			
226 			// loop through all laps in this activity
227 			for (var j = 0; j < lapNodes.length; j++) {
228 				
229 				// update the duration of this activity
230 				var lapTotalTime = Garmin.TcxActivityFactory._tagValue(lapNodes[j], Garmin.TcxActivityFactory.SCHEMA_TAGS.lapTotalTime);
231 				activityDurationMS += parseFloat(lapTotalTime + "e+3");
232 				
233 				/* not implemented until sections are in place
234 				// create lap section
235 				// set start time				
236 				// set total time				
237 				// set distance				
238 				// set max speed				
239 				// set calories				
240 				// set intensity				
241 				// set trigger method
242 				*/
243 				
244 				// loop through all the tracks in this lap
245 				var trackNodes = lapNodes[j].getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.track);			
246 				for (var k = 0; k < trackNodes.length; k++) {
247 					
248 					/* not implemented until sections are in place
249 					// create track section
250 					*/					
251 					
252 					// loop through all the trackpoints in this track
253 					var trackPointNodes = trackNodes[k].getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.trackPoint);
254 					for (var l = 0; l < trackPointNodes.length; l++) {
255 						//historySeries.addSample(Garmin.TcxActivityFactory._parseTcxTrackPoint(trackPointNodes[l]));
256 						var trackPoint = new Garmin.Sample();
257 						trackPoint.setLazyLoading(true, Garmin.TcxActivityFactory, trackPointNodes[l]);
258 						historySeries.addSample(trackPoint);
259 						//historySeries.addSample(new Garmin.Sample());
260 					}					
261 				}
262 			}
263 			
264 			if ( lapNodes.length > 0) {
265 				// set the start and end time summary data for the activity if possible
266 				activityStartTimeObj = (new Garmin.DateTimeFormat()).parseXsdDateTime(activityStartTimeMS);
267 				activityEndTimeObj	=  new Garmin.DateTimeFormat();
268 				// NOTE: switch to using setDate() once it is implemented in Garmin.DateTimeFormat
269 				activityEndTimeObj.date = new Date(activityStartTimeObj.getDate().getTime() + activityDurationMS);
270 				activity.setSummaryValue(Garmin.Activity.SUMMARY_KEYS.startTime, activityStartTimeObj);
271 				activity.setSummaryValue(Garmin.Activity.SUMMARY_KEYS.endTime, activityEndTimeObj);
272 			}
273 			
274 			if (historySeries.getSamplesLength() > 0) {				
275 				// add the populated series to the activity
276 				activity.addSeries(historySeries);
277 			}
278 			
279 			// Add the populated activity to the list of activities.  This activity may not have laps (if it's a directory listing entry).
280 			activities.push(activity);
281 		}
282 		
283 		return activities;

360 	_parseTcxActivity: function(activityNode, activityType) {
361 		// create new activity object
362 		var activity = new Garmin.Activity();
363 		
364 		// set lazy loaded
365 		activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.isLazyLoaded, true);
366 		
367 		// set factory
368 		activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.factory, Garmin.TcxActivityFactory);
369 		
370 		// set dom
371 		activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.dom, activityNode);
372 		
373 		// set id
374 		var id;
375 		if(activityType == Garmin.TcxActivityFactory.SCHEMA_TAGS.activity) {
376 			id = Garmin.TcxActivityFactory._tagValue(activityNode, Garmin.TcxActivityFactory.SCHEMA_TAGS.activityId);
377 		} else {
378 			id = Garmin.TcxActivityFactory._tagValue(activityNode, Garmin.TcxActivityFactory.SCHEMA_TAGS.courseName);
379 		}
380 		activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.activityName, id)		
381 		
382 		// set sport
383 		var sport = activityNode.getAttribute(Garmin.TcxActivityFactory.SCHEMA_TAGS.activitySport);
384 		activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.activitySport, sport);	
385 		
386 		// set creator information, optional in schema
387 		var creator = activityNode.getElementsByTagName(Garmin.TcxActivityFactory.SCHEMA_TAGS.creator);
388 		if (creator != null && creator.length > 0) {
389 			// set creator name
390 			var creatorName = Garmin.TcxActivityFactory._tagValue(creator[0], Garmin.TcxActivityFactory.SCHEMA_TAGS.creatorName);
391 			activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.creatorName, creatorName);
392 			
393 			// set creator unit id
394 			var unitId = Garmin.TcxActivityFactory._tagValue(creator[0], Garmin.TcxActivityFactory.SCHEMA_TAGS.creatorUnitID);
395 			activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.creatorUnitId, unitId);
396 							
397 			// set creator product id
398 			var prodId = Garmin.TcxActivityFactory._tagValue(creator[0], Garmin.TcxActivityFactory.SCHEMA_TAGS.creatorProductID);
399 			activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.creatorProdId, prodId);
400 							
401 			// set creator version
402 			var version = Garmin.TcxActivityFactory._parseTcxVersion(creator[0]);
403 			if (version != null) {
404 				activity.setAttribute(Garmin.Activity.ATTRIBUTE_KEYS.creatorVersion, version);
405 			}
406 		}
407 		
408 		return activity;
