 22 /** A place-holder for Garmin device information.
 23  * @class Garmin.Device
 24  *
 25  * @constructor 
 26  * @param {String} displayName for the device
 27  * @param {Number} number of the model
 28  */
 29 Garmin.Device = function(displayName, number){}; //just here for jsdoc
 30 Garmin.Device = Class.create();
 31 Garmin.Device.prototype = {
 32 
 33 	initialize: function(displayName, number) {
 34 	    this.displayName = displayName;
 35 	    this.number = number;
 36 	    this.parentNumber = -1;
 37 	    this.parent = null;
 38 	    this.children = new Array();
 39 	    
 40 	    this.partNumber = null;
 41 	    this.softwareVersion = null;
 42 	    this.description = null;
 43 	    this.id = null;
 44 	    this.fileBased = false;
 45 	    this.dataTypes = new Hash({});
 46 	},
 47 	
 48 	/** The display name of this device.
 49 	 * @type String
 50 	 * @return display name
 51 	 * @member Garmin.Device
 52 	 */
 53 	getDisplayName: function() {
 54 		return this.displayName;
 55 	},
 56 	
 57 	/** The device number that the plug-in uses to identify this device
 58 	 * @type Number
 59 	 * @return device number
 60 	 */
 61 	getNumber: function() {
 62 		return this.number;
 63 	},
 64 	
 65 	/** The parent device's number. -1 if this device has no parent.
 66 	 * @returns {Number} parent device number
 67 	 */
 68 	getParentNumber: function() {
 69 		return this.parentNumber;
 70 	},
 71 
 72 	/** Set parent device number
 73 	 * @param {Number} parent device number
 74 	 */
 75 	setParentNumber: function(aNumber) {
 76 		this.parentNumber = aNumber;
 77 	},
 78 
 79 	/** The parent device object, null if this device has no parent.
 80 	 * @returns {Garmin.Device} parent device
 81 	 */
 82 	getParent: function() {
 83 		return this.parentDevice;
 84 	},
 85 
 86 	/** Set parent device
 87 	 * @param {Garmin.Device} parent device object
 88 	 */
 89 	setParent: function(aDevice) {
 90 		this.parentDevice = aDevice;
 91 	},
 92 
 93 	/** Add a child device
 94 	 * @param {Garmin.Device} child object
 95 	 */
 96 	addChild: function(aDevice) {
 97 		this.children.push(aDevice);
 98 	},
 99     
100 	/** The array of child devices
101 	 * @returns {Array} of Garmin.Device objects
102 	 */
103     getChildren: function() {
104         return this.children;
105     },
106 
107 	/** Set part number of device
108 	 * @param {String} part number
109 	 */
110 	setPartNumber: function(partNumber) {
111 		this.partNumber = partNumber;
112 	},
113 
114 	/** The part number of the device
115 	 * @type String
116 	 * @return The part number of the device
117 	 */
118 	getPartNumber: function() {
119 		return this.partNumber;
120 	},
121 
122 	/** Software version installed on device
123 	 * @param {String} Garmin.Device
124 	 */
125 	setSoftwareVersion: function(softwareVersion) {
126 		this.softwareVersion = softwareVersion;
127 	},
128 
129 	/** The software version currently on the device
130 	 * @type String
131 	 * @return software version
132 	 */
133 	getSoftwareVersion: function() {
134 		return this.softwareVersion;
135 	},
136 
137 	/** Set description of the device
138 	 * @param {String} device description
139 	 */
140 	setDescription: function(description) {
141 		this.description = description;
142 	},
143 
144 	/** A description of the device.  This usually represents the model name of the device
145 	 * and includes the software version, i.e. "Forerunner 405  Jun 27 2008 2.17"
146 	 * In the GarminDevice XML, this is Model > Description.
147 	 * @type String
148 	 * @return device description
149 	 */
150 	getDescription: function() {
151 		return this.description;
152 	},
153 
154 	/** set device id
155 	 * @param {String} device id
156 	 */
157 	setId: function(id) {
158 		this.id = id;
159 	},
160 
161 	/** The device id
162 	 * @type String
163 	 * @return The device id
164 	 */
165 	getId: function() {
166 		return this.id;
167 	},
168 	
169 	/** Adds a data type to the list of data types supported by this device
170 	 * @param dataType A DeviceDataType object containing information about the data type being added
171 	 */
172 	addDeviceDataType: function(dataType) {
173 		var newDataType = new Hash();
174 		newDataType.set(dataType.getTypeName(), dataType);
175 		this.dataTypes.update(newDataType);	
176 	},
177 
178 	/** Returns a specific DeviceDataType object
179 	 * @type Garmin.DeviceDataType
180 	 * @return The DeviceDataType object containing the specified extension
181 	 * @param extension The file extension of the data type you are trying to get
182 	 */	
183 	getDeviceDataType: function(extension) {
184 	    return this.dataTypes.get(extension);
185 	},
186 
187 	/** Returns a hash containing all DeviceDataType objects
188 	 * @type Hash
189 	 * @return all DeviceDataType objects
190 	 */	
191 	getDeviceDataTypes: function() {
192 		return this.dataTypes;
193 	},
194 
195 	/**	Returns true if the device has read support for the file type
196 	 * @param {String} extension The extension of the file type you are checking for support
197 	 * @type Boolean
198 	 * @return read support for the file type
199 	 */	
200 	supportDeviceDataTypeRead: function(extension) {
201 		var dataType = this.getDeviceDataType(extension);
202 		if (dataType != null && dataType.hasReadAccess()) {
203 			return true;
204 		} else {
205 			return false;
206 		}	
207 	},
208 	
209 	/** Check if device has write support for the file type.
210 	 * @param {String} extension The extension of the file type you are checking for support
211 	 * @type Boolean
212 	 * @return True if write support 
213 	 */		
214 	supportDeviceDataTypeWrite: function(extension) {
215 		var dataType = this.getDeviceDataType(extension);
216 		if (dataType != null && dataType.hasWriteAccess()) {
217 			return true;
218 		} else {
219 			return false;
220 		}			
221 	},
222 
223 	/**
224 	 * Set if device is a File-based device.
225      * @param {Boolean} set if device is file based
226      */
227 	setIsFileBased: function(aBool) {
228 	   this.fileBased = aBool;	
229 	},
230 
231 	/**
232 	 * Check if device is a File-based device.<br\>
233 	 * Will return false for all devices on plug-in versions prior to 2.8.1.0 <br\>
234 	 * @see Garmin.DevicePlugin#isDeviceFileBased
235      * @returns {Boolean} true if device is file based
236      */
237 	isFileBased: function() {
238 	   return this.fileBased;	
239 	},
240 	
241 	toString: function() {
242 		return "Device["+this.getDisplayName()+", "+this.getDescription()+", "+this.getNumber()+"]";
243 	}
244 	
245 };

247 /** A place-holder for Garmin Device Data Type information
248  * @class Garmin.DeviceDataType
249  *
250  * @constructor 
251  * @param {String} typeName for the data type
252  * @param {String} extension file extension for the data type
253  */
254 Garmin.DeviceDataType = function(typeName, fileExtension){}; //just here for jsdoc
255 Garmin.DeviceDataType = Class.create();
256 Garmin.DeviceDataType.prototype = {
257 	
258 	initialize: function(typeName, fileExtension) {
259 		this.typeName = typeName;
260 		this.fileExtension = fileExtension;
261 		this.readAccess = false;
262 		this.writeAccess = false;
263 		this.filePath = null;
264 		this.readFilePath = null;
265 		this.writeFilePath = null;
266 		this.identifier = null;
267 	},
268 	
269 	/**
270 	 * @type String
271 	 * @return The type name of this data type
272 	 */
273 	getTypeName: function() {
274 		return this.typeName;
275 	},
276 	
277 	/** 
278 	 * @deprecated
279 	 * @type String
280 	 * @return The type/display name of this data type
281 	 */
282 	getDisplayName: function() {
283 		return this.getTypeName();
284 	},
285 	
286 	/**
287 	 * @type String
288 	 * @return The file extension of this data type
289 	 */
290 	getFileExtension: function() {
291 		return this.fileExtension;
292 	},
293 	
294 	/**
295 	 * @type String
296 	 * @return The file path for this data type
297 	 */
298 	getFilePath: function() {
299 		return this.filePath;
300 	},
301 	
302 	/**
303 	 * @param filePath - the filepath for this datatype
304 	 */
305 	setFilePath: function(filePath) {
306 		this.filePath = filePath;
307 	},
308 	
309 	/**
310      * @param readFilePath - the readFilePath for this datatype
311      */
312 	getReadFilePath: function() {
313 	   return this.readFilePath;	
314 	},
315 	
316 	/**
317 	 * @type String
318      * @return The read file path for this data type
319      */
320 	setReadFilePath: function(readFilePath) {
321 		this.readFilePath = readFilePath;
322 	},
323 	
324 	/**
325      * @param writeFilePath - the readFilePath for this datatype
326      */
327     getWriteFilePath: function() {
328        return this.writeFilePath;    
329     },
330     
331     /**
332      * @type String
333      * @return The write file path for this data type
334      */
335     setWriteFilePath: function(writeFilePath) {
336         this.writeFilePath = writeFilePath;
337     },
338 	
339 	/**
340 	 * @type String
341 	 * @return The identifier for this data type
342 	 */
343 	getIdentifier: function() {
344 		return this.identifier;
345 	},
346 	
347 	/**
348 	 * @param identifier- the identifier for this datatype
349 	 */
350 	setIdentifier: function(identifier) {
351 		this.identifier = identifier;
352 	},
353 	
354 	/**
355 	 * @param readAccess True == has read access
356 	 */
357 	setReadAccess: function(readAccess) {
358 		this.readAccess = readAccess;
359 	},
360 	
361 	/** Returns value indicating if the device supports read access for this file type
362 	 * @type Boolean
363 	 * @return supports read access for this file type
364 	 */
365 	hasReadAccess: function() {
366 		return this.readAccess;
367 	},
368 	
369 	/**
370 	 * @param {Boolean} has write access
371 	 */	
372 	setWriteAccess: function(writeAccess) {
373 		this.writeAccess = writeAccess;
374 	},
375 	
376 	/** return the value indicating if the device supports write access for this file type
377 	 * @type Boolean
378 	 * @return supports write access for this file type
379 	 */
380 	hasWriteAccess: function() {
381 		return this.writeAccess;
382 	}	
383 };
