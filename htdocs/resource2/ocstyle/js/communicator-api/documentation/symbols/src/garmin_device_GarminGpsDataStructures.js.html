363 
364     initialize: function() {
365         this.points = new Array();
366     },
367     
368     addTrackPoint: function(trackPointObject) {
369     	this.points.push(trackPointObject);
370     },
371     
372     /** Find the nearest valid point to the index given
373      * 
374      * @param index is the index
375      * @param incDirection is an int in the direction we'd like to look positive 
376      * 	nums are forward, negative nums are backwards
377      * 
378      * @type Garmin.TrackPoint 
379      * @return The nearest point (possibly the index) that has a validLocation
380      */ 
381     findNearestValidLocationPoint: function(index, incDirection) {
382         if( this.getPoint( index ).isValidLocation() ) {
383             return this.getPoint( index );
384         } else if( index >= this.getLength() ) {
385         	return this.findNearestValidLocationPoint(this.getLength()-1, -1);
386         } else {
387             return this.findNearestValidLocationPoint(index+incDirection, incDirection);
388         }
389     },

391 	/** Get the point specified on the track
392 	 * If the number is negative, get's the first
393 	 * If it's larger than possible, get's the last
394 	 * Otherwise it gets the number requested
395 	 *
396 	 * @param {Number} index is the point we want
397      * @type Garmin.TrackPoint 
398 	 * @return A TrackPoint that fits the pattern described above 
399 	 */
400     getPoint: function(index) {
401         index = Math.floor(index);
402     
403         if(index >= this.getLength()) {
404             return this.getEnd();
405         }
406         if(index <= 0) {
407             return this.getStart();
408         }
409             
410         return this.points[index];
411     },

703     /** Parse a GPX xml document for tracks
704 	 * @param {Document} xml document in GPX format
705      * @type Array<Garmin.Track>
706      */
707 	parseGpxTracks: function(gpxDocument) {
708 		var tracks = new Array();
709 
710     	var trackNodes = gpxDocument.getElementsByTagName("trk");
711 
712 		// triple for-loop fun
713 		for( var i=0; i < trackNodes.length; i++ ) {
714 			var trk = new Garmin.Track();
715 
716 			var trackSegments = trackNodes[i].getElementsByTagName("trkseg");
717 	
718 			for( var j=0; j < trackSegments.length; j++ ) {
719 				var trkseg = new Garmin.TrackSegment();
720 				
721 				var trackPoints = trackSegments[j].getElementsByTagName("trkpt");
722 		
723 				for( var k=0; k < trackPoints.length; k++ ) {
724 					var trkpt = new Garmin.TrackPoint();
725 
726 					var lat = trackPoints[k].getAttribute("lat");
727 					var lng = trackPoints[k].getAttribute("lon");
728 					var eleElement = trackPoints[k].getElementsByTagName("ele");
729 					var ele = (eleElement.length > 0) ? eleElement[0].childNodes[0].nodeValue : null;
730 					
731 					var timeNodes = trackPoints[k].getElementsByTagName("time");
732 					if(timeNodes.length > 0) {
733 						var time = timeNodes[0].childNodes[0].nodeValue;
734 						trkpt.date = (new Garmin.DateTimeFormat()).parseXsdDateTime(time);
735 					}
736 					
737 					trkpt.measurements = {
738 						latitude: {
739 							value: lat,
740 							context: "latitude"
741 						},
742 						longitude: {
743 							value: lng,
744 							context: "longitude"
745 						},
746 						elevation: {
747 							value: ele,
748 							context: "feet"
749 						}
750 					};
751 					
752 					trkseg.addTrackPoint(trkpt);
753 				}
754 				
755 				trk.addSegment(trkseg);
756 			}
757 
758 			tracks.push(trk);
759 		}
760 
761     	this.tracks = tracks;
762     	return tracks;
763 	},

765     /** Parse a GPX xml waypoint into a Waypoint object.
766 	 * @param {Element} xml waypoint in GPX format
767      * @type Garmin.WayPoint
768      */
769 	parseGpxWaypoint: function(waypointNode) {
770 		var lat  = waypointNode.getAttribute("lat");
771 		var lng  = waypointNode.getAttribute("lon");
772 		var name = this._tagValue(waypointNode,"name");
773 		var desc = this._tagValue(waypointNode,"desc");
774 		var ele  = this._tagValue(waypointNode,"ele");
775 		var sym  = this._tagValue(waypointNode,"sym");
776 		var type  = this._tagValue(waypointNode,"type");
777 		var cmt  = this._tagValue(waypointNode,"cmt");
778 		
779 		var wpt  = new Garmin.WayPoint(lat, lng, ele, name, null, desc, sym, type, cmt);		
780    		return wpt;

799     /** Take a list of tracks and waypoints and generate a GPX xml string
800 	 * @param {Array<Garmin.Track>} Tracks
801 	 * @param {Array<Garmin.WayPoint>} Waypoints
802      * @type String
803      */
804 	produceGpxString: function(tracks, waypoints) {
805 		gpxString = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"no\" ?>";
806 	
807 		gpxString += "<gpx xmlns=\"http://www.topografix.com/GPX/1/1\" xmlns:gpxx=\"http://www.garmin.com/xmlschemas/GpxExtensions/v3\" creator=\"Garmin Communicator Plug-In API\" version=\"1.1\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://www.topografix.com/GPX/1/1 http://www.topografix.com/GPX/1/1/gpx.xsd http://www.garmin.com/xmlschemas/GpxExtensions/v3 http://www.garmin.com/xmlschemas/GpxExtensions/v3/GpxExtensionsv3.xsd\">";
808 		
809 		if(tracks != null) {
810 			for( var i=0; i < tracks.length; i++ ) {
811 				gpxString += this.produceTrackGpxString(tracks[i]);
812 			}
813 		}
814 
815 		if(waypoints != null) {		
816 			for( var i=0; i < waypoints.length; i++ ) {
817 				gpxString += this.produceWaypointGpxString(waypoints[i]);
818 			}
819 		}
820 		gpxString += "</gpx>";
821 	
822 		return gpxString;
823 	},

825     /** Take a track object and generate a GPX xml string (without gpx or xml headers)
826 	 * @param {Garmin.Track} Track
827      * @type String
828      */
829 	produceTrackGpxString: function(track) {
830 		gpxString = "<trk>";
831 
832 		for( var i=0; i < track.getNumSegments(); i++ ) {
833 			var segment = track.getSegment(i);
834 			
835 			gpxString += "<trkseg>";
836 			for(var j=0; j < segment.getLength(); j++) {
837 				var point = segment.getPoint(j);
838 				
839 				gpxString += "<trkpt lat=\"" + point.getLat() + "\" lon=\"" + point.getLng() + "\">";
840 				if(point.getElev()) {
841 					gpxString += "<ele>" + point.getElev() + "</ele>";
842 				}
843 				if(point.getDate()) {
844 					gpxString += "<time>" + point.getDate().getXsdString() + "</time>";
845 				}
846 				gpxString += "</trkpt>";
847 			}
848 			gpxString += "</trkseg>";
849 
850 		}
851 		
852 		gpxString += "</trk>";
853 	
854 		return gpxString;
855 	},

857     /** Take a waypoint object and generate a GPX xml string (without gpx or xml headers)
858 	 * @param {Garmin.WayPoint} WayPoint
859      * @type String
860      */
861 	produceWaypointGpxString: function(waypoint) {
862 		gpxString = "<wpt lat=\"" + waypoint.getLat() + "\" lon=\"" + waypoint.getLng() + "\">";
863 	
864 		if(waypoint.getElev()) {
865 			gpxString += "<ele>" + waypoint.getElev() + "</ele>";
866 		}
867 		if(waypoint.getName()) {
868 			gpxString += "<name>" + waypoint.getName() + "</name>";
869 		}
870 		if(waypoint.getComment()) {
871 			gpxString += "<cmt>" + waypoint.getComment() + "</cmt>";
872 		}
873 		if(waypoint.getDescription()) {
874 			gpxString += "<desc>" + waypoint.getDescription() + "</desc>";
875 		}
876 		if(waypoint.getSymbol()) {
877 			gpxString += "<sym>" + waypoint.getSymbol() + "</sym>";
878 		}
879 		if(waypoint.getAddress()) {
880 			gpxString += "<extensions><gpxx:WaypointExtension>" + waypoint.getAddress() + "</gpxx:WaypointExtension></extensions>";
881 		}
882 		if(waypoint.getType()) {
883 			gpxString += "<type>" + waypoint.getType() + "</type>";
884 		}
885 		gpxString += "</wpt>";
886 	
887 		return gpxString;
888 	},
