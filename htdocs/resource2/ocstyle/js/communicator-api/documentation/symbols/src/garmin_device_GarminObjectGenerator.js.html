 21 /**
 22  * This snippet of code is required to generate the object tag
 23  * only when the browser reports that the plugin is installed.
 24  * 
 25  * @requires PluginDetect, BrowserDetect
 26  */
 27 if (PluginDetect.detectGarminCommunicatorPlugin()) {
 28 	
 29 	// Insert object tag based on browser
 30 	switch(BrowserDetect.browser) {
 31 	    // TODO pull these out into constants later, in BrowserDetect
 32 	    case "Explorer":
 33             document.write('<object id="GarminActiveXControl" style="WIDTH: 0px; HEIGHT: 0px; visible: hidden" height="0" width="0" classid="CLSID:099B5A62-DE20-48C6-BF9E-290A9D1D8CB5"> </object>');
 34             break;
 35 	    case "Firefox":
 36 	    case "Mozilla":
 37 	    case "Safari":
 38         	// Outer div necessary for Safari and Chrome
 39         	// TODO try removing the divs to see if Safari 3+ has fixed their bug  
 40         	document.write('<div style="height:0px; width:0px;">');
 41         	document.write('<object id="GarminNetscapePlugin" type="application/vnd-garmin.mygarmin" width="0" height="0"> </object>');
 42         	document.write('</div>');
 43             break;
 44 	    default:
 45         	document.write('<div style="height:0px; width:0px;"><object id="GarminActiveXControl" style="WIDTH: 0px; HEIGHT: 0px; visible: hidden" height="0" width="0" classid="CLSID:099B5A62-DE20-48C6-BF9E-290A9D1D8CB5">');
 46         	document.write('	<object id="GarminNetscapePlugin" type="application/vnd-garmin.mygarmin" width="0" height="0"> </object>');
 47         	document.write('</object></div>');
 48         	break;
 49 	}
 50 	
 51 }
