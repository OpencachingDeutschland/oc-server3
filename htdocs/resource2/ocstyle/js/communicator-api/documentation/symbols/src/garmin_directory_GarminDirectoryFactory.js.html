 20 /**A factory that can produce a Garmin.Directory object given directory xml
 21  * conforming to schema: <a href="http://www.garmin.com/xmlschemas/DirectoryListingV1.xsd">http://www.garmin.com/xmlschemas/DirectoryListingV1.xsd</a>
 22  * @class Garmin.DirectoryFactory
 23  * @constructor 
 24  * @requires Garmin.File, Garmin.FileId
 25  */
 26 Garmin.DirectoryFactory = function(){};
 27 Garmin.DirectoryFactory = {
 28 	
 29 	parseString: function(xmlString) {
 30 		var dirDocument = Garmin.XmlConverter.toDocument(xmlString);		
 31 		return Garmin.DirectoryFactory.parseDocument(dirDocument);		
 32 	},
 33 	
 34 	/* Creates and returns a list of all files from the document, regardless of type. */
 35 	parseDocument: function(dirDocument) {
 36 		
 37 		// Not parseable directory doc
 38 		if( dirDocument.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.directoryListing).length == 0) {
 39 			throw new Error("ERROR: Unable to parse directory document.");
 40 		}
 41 		
 42 		var parsedDocument;
 43 		
 44 		// Files		
 45 		if( dirDocument.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.file).length >= 0) {
 46 			// Complete file
 47 			parsedDocument = Garmin.DirectoryFactory._parseFiles(dirDocument);
 48 		} 
 49 		
 50 		return parsedDocument;
 51 	},
 52 	
 53 	produceString: function(directory) {
 54 		// TODO fill this guy out when we need it
 55 	},
 56 	
 57 	_parseFiles: function(dirDocument) {
 58 		var files = new Array();
 59 		var fileNodes;
 60 
 61 		// Grab the activity/course nodes, depending on document		
 62 		fileNodes = dirDocument.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.file);
 63 		
 64 		// loop through all files in the document
 65 		for (var i = 0; i < fileNodes.length; i++) {
 66 			
 67 			// create new file object
 68 			var file = Garmin.DirectoryFactory._parseFile(fileNodes[i], Garmin.DirectoryFactory.SCHEMA_TAGS.file);
 69 			
 70 			// Add the populated file to the list of files.
 71 			files.push(file);
 72 		}
 73 		
 74 		return files;
 75 	},
 76 	
 77 	_parseFile: function(fileNode, fileType) {
 78 		// create new activity object
 79 		var file = new Garmin.File();
 80 		
 81 		// set is directory
 82 		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.isDirectory, fileNode.getAttribute(Garmin.File.ATTRIBUTE_KEYS.isDirectory));
 83 		
 84 		// set path
 85 		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.path, fileNode.getAttribute(Garmin.File.ATTRIBUTE_KEYS.path));
 86 		
 87 		// set type
 88 		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.type, fileNode.getAttribute(Garmin.File.ATTRIBUTE_KEYS.type));
 89 
 90 		// set creation time, optional.
 91 		var creationTimeNode = fileNode.getElementsByTagName(Garmin.File.ATTRIBUTE_KEYS.creationTime)[0];
 92 		if( creationTimeNode != null ) {
 93     		var creationTime = creationTimeNode.childNodes[0].nodeValue;
 94     		var creationTimeObj = (new Garmin.DateTimeFormat()).parseXsdDateTime(creationTime);
 95     		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.creationTime, creationTimeObj);
 96 		}
 97         
 98         //set size, optional
 99 		var theSize = fileNode.getAttribute(Garmin.File.ATTRIBUTE_KEYS.size);
100 		if( theSize != null ) {
101 		    theSize = theSize * 1; //convert to number
102     		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.size, theSize);
103 		}
104         
105         //set checksum, optional
106 		var theMD5 = fileNode.getAttribute(Garmin.File.ATTRIBUTE_KEYS.md5Checksum);
107 		if( theMD5 != null ) {
108     		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.md5Checksum, theMD5);
109 		}
110 
111         // set dom
112 		file.setAttribute(Garmin.File.ATTRIBUTE_KEYS.dom, fileNode);
113 		
114 		// set fit id - only one id per file	
115 		// TODO Will there be other id types in the future? probably... 	
116 		var fitIdNode = fileNode.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.fitId)[0];
117 		if( fitIdNode )
118 		{
119 			var fitId = Garmin.DirectoryFactory._parseFitId(fitIdNode);
120 			file.setId(fitId);
121 		}
122 		return file;
123 	},
124 	
125 	_parseFitId: function(fitIdNode) {
126 	    var fitId = new Garmin.FileId();
127 	     
128 		// set id
129 		var id = Garmin.DirectoryFactory._tagValue(fitIdNode, Garmin.DirectoryFactory.SCHEMA_TAGS.id);
130 		if( id != null ) {
131 		  fitId.setValue(Garmin.DirectoryFactory.SCHEMA_TAGS.id, id);
132 		}
133         // set filetype
134 		var fileType = Garmin.DirectoryFactory._tagValue(fitIdNode, Garmin.DirectoryFactory.SCHEMA_TAGS.fileType);
135 		if( fileType != null ) {
136 		  fitId.setValue(Garmin.DirectoryFactory.SCHEMA_TAGS.fileType, fileType);
137 		}
138         // set manufacturer
139 		var manufacturer = Garmin.DirectoryFactory._tagValue(fitIdNode, Garmin.DirectoryFactory.SCHEMA_TAGS.manufacturer);
140 		if( manufacturer != null ) {
141 		  fitId.setValue(Garmin.DirectoryFactory.SCHEMA_TAGS.manufacturer, manufacturer);
142 		}
143         // set product
144 		var product = Garmin.DirectoryFactory._tagValue(fitIdNode, Garmin.DirectoryFactory.SCHEMA_TAGS.product);
145 		if( product != null ) {
146 		  fitId.setValue(Garmin.DirectoryFactory.SCHEMA_TAGS.product, product);
147 		}
148         // set serial number
149 		var serialNumber= Garmin.DirectoryFactory._tagValue(fitIdNode, Garmin.DirectoryFactory.SCHEMA_TAGS.serialNumber);
150 		if( serialNumber != null ) {
151 		  fitId.setValue(Garmin.DirectoryFactory.SCHEMA_TAGS.serialNumber, serialNumber);
152 		}
153 		
154 		return fitId;
155 	},
156 	
157 	/**
158 	 * Takes in a list of any file type and returns a list of only activity file types.
159 	 * @param files {Array} list of Garmin.File objects of any type produced from the factory 
160 	 */
161 	getActivityFiles: function(files) {
162 	    var activityFiles = new Array();
163 	    
164 	    for(var i=0; i < files.length; i++) {
165 	        if( files[i].getIdValue(Garmin.FileId.KEYS.fileType) == Garmin.FileId.FILE_TYPE_MAP.activities)
166 	        {
167 	            activityFiles.push(files[i]);
168 	        }
169 	    }
170 	    
171 	    return activityFiles;
172 	},
173 	
174 	/**
175 	 * Takes in a list of any file type and returns a list of only health data file types.
176 	 * @param files {Array} list of Garmin.File objects of any type produced from the factory 
177 	 */
178 	getHealthDataFiles: function(files) {
179 	    var activityFiles = new Array();
180 	    
181 	    for(var i=0; i < files.length; i++) {
182 	        if( files[i].getIdValue(Garmin.FileId.KEYS.fileType) == Garmin.FileId.FILE_TYPE_MAP.monitoring)
183 	        {
184 	            activityFiles.push(files[i]);
185 	        }
186 	    }
187 	    
188 	    return activityFiles;
189 	},
190 	
191 	/**
192 	 * Gets the first tag's value under the parent.
193 	 */
194 	_tagValue: function(parentNode, tagName) {
195 		var subNode = parentNode.getElementsByTagName(tagName);
196 		return subNode.length > 0 ? subNode[0].childNodes[0].nodeValue : null;
197 	},	
198 	
199     toString: function() {
200         return "[DirectoryFactory]";
201     }	
202 };
