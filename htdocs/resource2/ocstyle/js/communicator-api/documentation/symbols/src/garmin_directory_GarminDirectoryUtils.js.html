 21 /**Utility functions for working with Garmin Directory Listing XML
 22  * conforming to schema: <a href="http://www.garmin.com/xmlschemas/DirectoryListingV1.xsd">http://www.garmin.com/xmlschemas/DirectoryListingV1.xsd</a>
 23  * @class Garmin.DirectoryUtils
 24  * @requires Garmin.File, Garmin.DirectoryFactory
 25  */
 26 Garmin.DirectoryUtils = function(){};
 27 Garmin.DirectoryUtils = {
 28    
 29    /**Merge the contents of two directory xml documents that
 30     * conform to schema: <a href="http://www.garmin.com/xmlschemas/DirectoryListingV1.xsd">http://www.garmin.com/xmlschemas/DirectoryListingV1.xsd</a>
 31 	* @param {String} aXmlString1 First directory XML string
 32 	* @param {String} aXmlString2 Second directory XML string
 33 	* @return {Document} the result of merging the two directory listing XML strings.
 34 	*/
 35 	merge: function(aXmlString1, aXmlString2) {
 36 							  
 37 		var theFirstDoc = Garmin.XmlConverter.toDocument(aXmlString1);
 38 		var theFirstDir = theFirstDoc.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.directoryListing);
 39 		var theSecondDoc = Garmin.XmlConverter.toDocument(aXmlString2);		
 40 		var theSecondDir = theSecondDoc.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.directoryListing);
 41 
 42 		if(theFirstDir.length > 0 && theSecondDir.length > 0) {
 43 			//Grab the file nodes
 44 			var theFileNodes = theSecondDoc.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.file);
 45 			var theDir = theFirstDir[0];
 46 			//loop through all files in the second document
 47 			for (var i = 0; i < theFileNodes.length; ++i) {
 48 				var theFileNode = Try.these(
 49 						function() { return theFirstDoc.importNode(theFileNodes[i],true);},
 50 						function() { return Garmin.DOM._importNode(theFirstDoc,theFileNodes[i],true);} 
 51 						);
 52 				if(!this._dirHasFileNode(theDir, theFileNode)) {
 53 					theDir.appendChild(theFileNode);
 54 				}
 55 			}
 56 		} else {
 57 		  theFirstDoc = null;
 58 		}
 59 		return theFirstDoc;	  
 60 	},						  
 61 	
 62 	/*
 63 	* Detect duplicate file nodes
 64 	* @private
 65 	*/
 66 	_dirHasFileNode: function(aDirNode, aFileNode) {
 67 		var theExists = false;
 68 		var theNodes = aDirNode.getElementsByTagName(Garmin.DirectoryFactory.SCHEMA_TAGS.file);
 69 		var theFilePath = aFileNode.getAttribute(Garmin.File.ATTRIBUTE_KEYS.path).toLowerCase();
 70 		for( var i = 0; i < theNodes.length; ++i )
 71 		{
 72 			var theChildPath = theNodes[i].getAttribute(Garmin.File.ATTRIBUTE_KEYS.path).toLowerCase();
 73 			if(!theChildPath)
 74 			{
 75 				continue;
 76 			}
 77 			if(theChildPath == theFilePath)
 78 			{
 79 				theExists = true;
 80 				break;
 81 			}
 82 		}		
 83 		return theExists;
 84 	},
 85 	
 86     toString: function() {
 87         return "[DirectoryUtils]";
 88     }	
 89 };

111 /*
112  * Slightly modified importNode implementation from of A List Apart for Internet Explorer
113  * Anthony T. Holdener III
114  * http://www.alistapart.com/d/crossbrowserscripting/xbImportNode.js
115  * @private
116  */
117 Garmin.DOM = function(){};
118 Garmin.DOM._importNode = function(aDoc, node, allChildren) {
119 	/* find the node type to import */
120 	switch (node.nodeType) {
121 		case document.ELEMENT_NODE:
122 			/* create a new element */
123 			var newNode = aDoc.createElement(node.nodeName);
124 			/* does the node have any attributes to add? */
125 			if (node.attributes && node.attributes.length > 0)
126 				/* add all of the attributes */
127 				for (var i = 0, il = node.attributes.length; i < il;)
128 					newNode.setAttribute(node.attributes[i].nodeName, node.getAttribute(node.attributes[i++].nodeName));
129 			/* are we going after children too, and does the node have any? */
130 			if (allChildren && node.childNodes && node.childNodes.length > 0)
131 				/* recursively get all of the child nodes */
132 				for (var i = 0, il = node.childNodes.length; i < il;)
133 					newNode.appendChild(Garmin.DOM._importNode(aDoc, node.childNodes[i++], allChildren));
134 			return newNode;
135 			break;
136 		case document.TEXT_NODE:
137 		case document.CDATA_SECTION_NODE:
138 		case document.COMMENT_NODE:
139 			return aDoc.createTextNode(node.nodeValue);
140 			break;
141 	}
142 };
